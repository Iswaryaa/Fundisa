<?php

include("configure.php"); // Using database connection file here

$id = $_GET['id']; // get id through query string

$qry = mysqli_query($db,"select * from enrollkid where id='$id'"); // select query

$data = mysqli_fetch_array($qry); // fetch data

if(isset($_POST['update'])) // when click on Update button
{
    $studentname = $_POST['studentname'];
    $age = $_POST['age'];
	
    $edit = mysqli_query($db,"update enrollkid set studentname='$studentname', age='$age' where id='$id'");
	
    if($edit)
    {
        mysqli_close($db); // Close connection
        header("location:kids-list.php"); // redirects to all records page
        exit;
    }
    else
    {
        echo mysqli_error();
    }    	
}
?>

<h3>Update Data</h3>

  <input type="submit" name="update" value="Update">
</form>
